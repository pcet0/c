import random
import hashlib
import math

# Check if number is prime (using built-in math.isqrt)
def is_prime(num):
    if num <= 1:
        return False
    for i in range(2, math.isqrt(num) + 1):
        if num % i == 0:
            return False
    return True

# Generate RSA key pair (public and private)
def generate_keypair():
    p = q = 1
    while not is_prime(p):
        p = random.randint(100, 1000)
    while not is_prime(q) or p == q:
        q = random.randint(100, 1000)

    n = p * q
    phi = (p - 1) * (q - 1)

    # Choose e using gcd from math
    e = random.randint(2, phi - 1)
    while math.gcd(e, phi) != 1:
        e = random.randint(2, phi - 1)

    # Use built-in modular inverse
    d = pow(e, -1, phi)

    return (e, n), (d, n)

# Encrypt message
def rsa_encrypt(message, public_key):
    e, n = public_key
    return [pow(ord(c), e, n) for c in message]

# Decrypt message
def rsa_decrypt(encrypted, private_key):
    d, n = private_key
    return ''.join(chr(pow(c, d, n)) for c in encrypted)

# Generate digital signature (hash + encrypt)
def generate_signature(message, private_key):
    hashed = hashlib.sha256(message.encode()).hexdigest()
    return rsa_encrypt(hashed, private_key)

# Verify digital signature (decrypt + compare)
def verify_signature(message, signature, public_key):
    hashed = hashlib.sha256(message.encode()).hexdigest()
    decrypted = rsa_decrypt(signature, public_key)
    return decrypted == hashed

# ---- Main Program ----
print("Generating keys for X (Sender) and Y (Receiver)...")
public_X, private_X = generate_keypair()
public_Y, private_Y = generate_keypair()

message = input("\nEnter message from X → Y: ")

encrypted = rsa_encrypt(message, public_Y)
signature = generate_signature(message, private_X)

print("\nEncrypted Message:", encrypted)
print("Signature:", signature)

# Simulated attacker
try:
    fake_key = "invalid"
    decrypted = rsa_decrypt(encrypted, fake_key)
    print("Z (attacker) read message:", decrypted)
except:
    print("\nZ (attacker) failed to decrypt message — not authorized.")

# Receiver side
decrypted_msg = rsa_decrypt(encrypted, private_Y)
verified = verify_signature(decrypted_msg, signature, public_X)

print("\nReceiver Side:")
print("Decrypted Message:", decrypted_msg)
if verified:
    print("Signature Verified — Message Integrity and Authenticity Confirmed.")
else:
    print("Signature Invalid — Message may have been tampered.")
