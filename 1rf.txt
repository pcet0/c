#include <iostream>
#include <string>
#include <vector>
using namespace std;

string encryptRailFence(const string &text, int key)
{
    if (key <= 1 || key >= text.size())
        return text;

    vector<string> rail(key, "");
    int row = 0, dir = 1;

    for (int i = 0; i < text.size(); i++)
    {
        rail[row] += text[i];

        if (row == 0)
            dir = 1;
        else if (row == key - 1)
            dir = -1;

        row += dir;
    }

    string result = "";
    for (int r = 0; r < key; r++)
        result += rail[r];

    return result;
}

int main()
{
    string text;
    int key;

    cout << "Rail Fence Cipher\n";

    cout << "Enter the plain text: ";
    getline(cin, text);

    cout << "Enter the key (number of rails): ";
    cin >> key;

    string encrypted = encryptRailFence(text, key);

    cout << "\nPlain Text: " << text << endl;
    cout << "Encrypted Text: " << encrypted << endl;

    return 0;
}
